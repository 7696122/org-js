org.js A parser and converter for org-mode notation

Repository on GitHub: http://github.com/mooz/org-js/

var code = [
  "* head1",
  "** head2",
  "- definition element ::",
  "    definition description",
].join("\n");
var parser = new Org.Parser();
var doc    = parser.parse(code);
var html   = Org.HtmlTextConverter.convertDocument(doc);
